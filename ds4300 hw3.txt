      _id: "$country_code",
      latest_cumulative_confirmed: {
        $sum: "$cumulative_confirmed"
      }
    }
  },
  {
    $lookup: {
      from: "demographics_normalized",
      let: { cc: "$_id" },
      pipeline: [
        {
          $match: {
            $expr: {
              $eq: ["$location_key", "$$cc"]
            }
          }
        }
      ],
      as: "demographics"
    }
  },
  {
    $addFields: {
      total_elderly: {
        $sum: [
          { $arrayElemAt: ["$demographics.population_age_60_69", 0] },
          { $arrayElemAt: ["$demographics.population_age_70_79", 0] },
          { $arrayElemAt: ["$demographics.population_age_80_and_older", 0] }
        ]
      },
      total_population: {
        $arrayElemAt: ["$demographics.population", 0]
      }
    }
  },
  {
    $addFields: {
      confirmed_per_capita: {
        $cond: {
          if: { $gt: ["$total_population", 0] },
          then: { $divide: ["$latest_cumulative_confirmed", "$total_population"] },
          else: null
        }
      },
      elderly_per_capita: {
        $cond: {
          if: { $gt: ["$total_population", 0] },
          then: { $divide: ["$total_elderly", "$total_population"] },
          else: null
        }
      }
    }
  },
  {
    $sort: {
      confirmed_per_capita: -1,
      elderly_per_capita: -1
    }
